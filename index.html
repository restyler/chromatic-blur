Usage Example

// Initialize with default settings
const blur = new ChromaticBlur('.navbar');

// Or with custom options
const blur = new ChromaticBlur('.navbar', {
  redOffset: 8,
  blueOffset: -8,
  blurAmount: 5,
  displacementScale: 75
});

// Update dynamically
blur.update({ redOffset: 10 });

// Method chaining
blur.disable().update({ blurAmount: 2 }).enable();

// Cleanup
blur.destroy();
